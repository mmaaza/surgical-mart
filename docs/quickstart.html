Option 1: cURL Quick Start

Step 1: Test Public Endpoint

Development

curl -X GET "http://localhost:5000/api/products?limit=5"

Production

curl -X GET "https://dentalkartnepal.com/api/products?limit=5"

Step 2: Register & Login

Register (Development)

curl -X POST "http://localhost:5000/api/auth/register" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Test User",
    "email": "test@example.com",
    "password": "password123"
  }'

Login (Development)

curl -X POST "http://localhost:5000/api/auth/login" \
  -H "Content-Type: application/json" \
  -d '{
    "email": "test@example.com",
    "password": "password123"
  }'

Response (Save the token!)

Option 3: JavaScript Quick Start

Setup API Client

// api-client.js
class MBNepalAPI {
  constructor(environment = 'development') {
    this.baseUrl = environment === 'production' 
      ? 'https://dentalkartnepal.com/api'
      : 'http://localhost:5000/api';
    this.token = localStorage.getItem('mbnepal_token');
  }

  async request(endpoint, options = {}) {
    const url = `${this.baseUrl}${endpoint}`;
    const headers = {
      'Content-Type': 'application/json',
      ...options.headers
    };

    if (this.token) {
      headers.Authorization = `Bearer ${this.token}`;
    }

    const response = await fetch(url, {
      ...options,
      headers
    });

    const data = await response.json();

    if (!response.ok) {
      throw new Error(data.error || 'API request failed');
    }

    return data;
  }

  async login(email, password) {
    const response = await this.request('/auth/login', {
      method: 'POST',
      body: JSON.stringify({ email, password })
    });

    if (response.success && response.token) {
      this.token = response.token;
      localStorage.setItem('mbnepal_token', this.token);
    }

    return response;
  }

  async getProducts(params = {}) {
    const query = new URLSearchParams(params).toString();
    return this.request(`/products${query ? '?' + query : ''}`);
  }

  async getProfile() {
    return this.request('/auth/me');
  }
}

// Usage
const api = new MBNepalAPI('development'); // or 'production'

Quick Test Script

// quick-test.js
async function quickTest() {
  try {
    // Test public endpoint
    console.log('Testing public endpoint...');
    const products = await api.getProducts({ limit: 5 });
    console.log('Products:', products.data.products);

    // Login
    console.log('Logging in...');
    const loginResult = await api.login('test@example.com', 'password123');
    console.log('Login successful:', loginResult.success);

    // Test protected endpoint
    console.log('Testing protected endpoint...');
    const profile = await api.getProfile();
    console.log('User profile:', profile.data);

  } catch (error) {
    console.error('Error:', error.message);
  }
}

// Run the test
quickTest();
